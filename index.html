                                    <p className="text-gray-800">{note.content}</p>

                                    {note.tag && <span className="text-xs text-blue-600 mt-2 block">#{note.tag}</span>}

        function TodosTab({ isOwner, todos, setTodos, workItems, todayStr }) {

                            const wItem = workItems.find(w => w.id === todo.workItemId);

                            const isDone = todo.status === 'Done';

                            const isOverdue = !isDone && todo.date < todayStr;

                            const isToday = !isDone && todo.date === todayStr;


                            let bgClass = 'bg-white';

                            if (isDone) bgClass = 'bg-gray-100 opacity-60';

                            else if (isOverdue) bgClass = 'bg-red-50 border-red-300';

                            else if (isToday) bgClass = 'bg-yellow-50 border-yellow-300';


                            return (

                                <div key={todo.id} className={`p-4 rounded shadow-sm border flex items-start gap-4 ${bgClass}`}>

                                    <input 

                                        type="checkbox" 

                                        checked={isDone} 

                                        onChange={() => toggleTodo(todo.id)} 

                                        disabled={!isOwner}

                                        className="mt-1 w-5 h-5 cursor-pointer"

                                    />

                                    <div className="flex-1">

                                        <div className="flex justify-between">

                                            <span className={`font-bold ${isDone ? 'line-through text-gray-500' : (isOverdue ? 'text-red-700' : 'text-gray-900')}`}>{todo.title}</span>

                                            <span className="text-xs text-gray-500">{todo.date}</span>

                                        </div>

                                        <p className="text-sm text-gray-600 mt-1">{todo.description}</p>

                                        <p className="text-xs text-blue-600 mt-2">{wItem ? wItem.title : ''}</p>

                                    </div>

                                    {isOwner && (

                                        <button onClick={() => setTodos(todos.filter(t => t.id !== todo.id))} className="text-red-500 text-xs">Sil</button>

                                    )}

                                </div>

                            );

                        })}

                        {todos.length === 0 && <p className="text-gray-500">Görev bulunamadı.</p>}

                    </div>

                </div>

            );

        }


        function AgendaTab({ todos, todayStr }) {

            // Simplified Weekly View simulation grouping by date

            const groupedTodos = todos.reduce((acc, todo) => {

                acc[todo.date] = acc[todo.date] || [];

                acc[todo.date].push(todo);

                return acc;

            }, {});


            const sortedDates = Object.keys(groupedTodos).sort();


            return (

                <div>

                    <h2 className="text-2xl font-bold mb-6">Ajanda Görünümü</h2>

                    <div className="grid grid-cols-1 md:grid-cols-3 gap-6">

                        {sortedDates.map(date => {

                            const isToday = date === todayStr;

                            return (

                                <div key={date} className={`p-4 rounded-lg border ${isToday ? 'bg-blue-50 border-blue-400' : 'bg-white shadow-sm'}`}>

                                    <h3 className={`font-bold border-b pb-2 mb-3 ${isToday ? 'text-blue-700' : 'text-gray-700'}`}>

                                        {date} {isToday && '(Bugün)'}

                                    </h3>

                                    <div className="flex flex-col gap-2">

                                        {groupedTodos[date].map(todo => (

                                            <div key={todo.id} className={`text-sm p-2 rounded ${todo.status === 'Done' ? 'line-through text-gray-400' : (date < todayStr ? 'bg-red-100 text-red-800' : 'bg-gray-100')}`}>

                                                {todo.title}

                                            </div>

                                        ))}

                                    </div>

                                </div>

                            )

                        })}

                        {sortedDates.length === 0 && <p className="text-gray-500">Ajandada kayıtlı iş yok.</p>}

                    </div>

                </div>

            );

        }


        function SummaryTab({ workItems, notes, todos, todayStr }) {

            

            const handleCopyRich = async () => {

                let htmlStr = '';

                let textStr = '';


                workItems.forEach(item => {

                    const itemNotes = notes.filter(n => n.workItemId === item.id && n.date <= todayStr).sort((a,b) => new Date(b.date) - new Date(a.date));

                    const latestNote = itemNotes.length > 0 ? itemNotes[0].content : 'Not bulunamadı.';

                    

                    htmlStr += `<p><b>${item.title}</b>-<b><a href="${item.jiraLink}">${item.jiraId}</a></b>-${latestNote}</p>`;

                    textStr += `${item.title}-${item.jiraId}-${latestNote}\n`;

                });


                try {

                    const blobHtml = new Blob([htmlStr], { type: 'text/html' });

                    const blobText = new Blob([textStr], { type: 'text/plain' });

                    const clipboardItem = new ClipboardItem({

                        'text/html': blobHtml,

                        'text/plain': blobText

                    });

                    await navigator.clipboard.write([clipboardItem]);

                    alert('Özet zengin metin (Rich Text) formatında kopyalandı! Mailinize veya dokümana yapıştırabilirsiniz.');

                } catch (err) {

                    await navigator.clipboard.writeText(textStr);

                    alert('Tarayıcınız zengin metin desteklemiyor. Düz metin olarak kopyalandı.');

                }

            };


            return (

                <div>

                    <div className="flex justify-between items-center mb-6">

                        <h2 className="text-2xl font-bold">Durum Özeti</h2>

                        <button onClick={handleCopyRich} className="bg-green-600 text-white px-4 py-2 rounded shadow hover:bg-green-700 font-bold">

                            Özeti Kopyala

                        </button>

                    </div>


                    <div className="flex flex-col gap-4">

                        {workItems.map(item => {

                            const itemNotes = notes.filter(n => n.workItemId === item.id);

                            const pastNotes = itemNotes.filter(n => n.date <= todayStr).sort((a,b) => new Date(b.date) - new Date(a.date));

                            const futureNotes = itemNotes.filter(n => n.date > todayStr);

                            

                            const latestNote = pastNotes.length > 0 ? pastNotes[0].content : 'Henüz not girilmemiş.';

                            const openTodos = todos.filter(t => t.workItemId === item.id && t.status !== 'Done');


                            return (

                                <div key={item.id} className="bg-white p-5 rounded-lg shadow-sm border border-gray-200">

                                    <div className="flex gap-2 items-center mb-2">

                                        <h3 className="font-bold text-lg">{item.title}</h3>

                                        <a href={item.jiraLink} target="_blank" className="bg-blue-100 text-blue-800 text-xs px-2 py-1 rounded font-bold hover:bg-blue-200">{item.jiraId}</a>

                                        <span className="text-xs bg-gray-100 px-2 py-1 rounded text-gray-600">{item.groupName}</span>

                                    </div>

                                    <p className="text-gray-800 bg-gray-50 p-3 rounded mb-3 border-l-4 border-gray-400">{latestNote}</p>

                                    

                                    {futureNotes.length > 0 && (

                                        <div className="mt-3 p-3 bg-purple-50 border border-purple-200 rounded">

                                            <h4 className="text-xs font-bold text-purple-800 mb-1">İleri Tarihli Notlar / Planlar</h4>

                                            <ul className="list-disc pl-4 text-sm text-purple-900">

                                                {futureNotes.map(fn => <li key={fn.id}>[{fn.date}] {fn.content}</li>)}

                                            </ul>

                                        </div>

                                    )}


                                    {openTodos.length > 0 && (

                                        <div className="mt-2 text-sm text-red-600 font-semibold">

                                            Bekleyen {openTodos.length} görev var.

                                        </div>

                                    )}

                                </div>

                            );

                        })}

                    </div>

                </div>

            );

        }
